cmake_minimum_required(VERSION 3.12)
project(Sofa.Component.PhysicalModel.SolidMechanics LANGUAGES CXX)

set(SOFACOMPONENTPHYSICALMODELSOLIDMECHANICS_SOURCE_DIR "src/sofa/component/physicalmodel/solidmechanics")

set(SOFACOMPONENTPHYSICALMODELSOLIDMECHANICS_DIRS Elasticity)
set(SOFACOMPONENTPHYSICALMODELSOLIDMECHANICS_TARGETS)
foreach(solidmechanics_dir ${SOFACOMPONENTPHYSICALMODELSOLIDMECHANICS_DIRS})
	add_subdirectory(${solidmechanics_dir})
	list(APPEND SOFACOMPONENTPHYSICALMODELSOLIDMECHANICS_TARGETS ${PROJECT_NAME}.${solidmechanics_dir})
endforeach()

set(HEADER_FILES
    ${SOFACOMPONENTPHYSICALMODELSOLIDMECHANICS_SOURCE_DIR}/config.h.in
)

set(SOURCE_FILES
    ${SOFACOMPONENTPHYSICALMODELSOLIDMECHANICS_SOURCE_DIR}/init.cpp
)

set(TARGET_NAME ${PROJECT_NAME})

configure_file(${SOFACOMPONENTPHYSICALMODELSOLIDMECHANICS_SOURCE_DIR}/config.h.in 
    ${CMAKE_BINARY_DIR}/include/sofa/component/physicalmodel/solidmechanics/config.h)

add_library(${TARGET_NAME} SHARED ${HEADER_FILES} ${SOURCE_FILES})
set_target_properties(${TARGET_NAME} PROPERTIES LINKER_LANGUAGE CXX)
target_link_libraries(${TARGET_NAME} PUBLIC ${SOFACOMPONENTPHYSICALMODELSOLIDMECHANICS_TARGETS})

target_compile_definitions(${TARGET_NAME} PRIVATE BUILD_SOFACOMPONENTPHYSICALMODELSOLIDMECHANICS)
target_include_directories(${TARGET_NAME} PUBLIC "$<BUILD_INTERFACE:${CMAKE_BINARY_DIR}/include>")
target_include_directories(${TARGET_NAME} PUBLIC "${CMAKE_CURRENT_SOURCE_DIR}/src/")
